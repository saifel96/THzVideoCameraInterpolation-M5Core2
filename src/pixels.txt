void apply_smoothing(float *data, uint8_t rows, uint8_t cols)
{
  float smoothed[rows * cols];
  for (int y = 0; y < rows; y++)
  {
    for (int x = 0; x < cols; x++)
    {
      float sum = 0;
      int count = 0;
      for (int dy = -1; dy <= 1; dy++)
      {
        for (int dx = -1; dx <= 1; dx++)
        {
          int nx = x + dx;
          int ny = y + dy;
          if (nx >= 0 && nx < cols && ny >= 0 && ny < rows)
          {
            sum += get_point(data, rows, cols, nx, ny);
            count++;
          }
        }
      }
      smoothed[y * cols + x] = sum / count;
    }
  }
  memcpy(data, smoothed, rows * cols * sizeof(float));
}
